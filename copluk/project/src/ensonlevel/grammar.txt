PROGRAM ::= STATEMENT_LIST EOF
STATEMENT_LIST ::= STATEMENT (SEMICOLON STATEMENT)*
STATEMENT ::= ASSIGN_STMT | IF_STMT | WHILE_STMT | EXPRESSION_STMT
ASSIGN_STMT ::= IDENTIFIER ASSIGN EXPRESSION SEMICOLON
IF_STMT ::= IF EXPRESSION THEN STATEMENT_LIST ELSE_STMT END //why expression ? 1 true 0 false? 
ELSE_STMT ::= ELSE STATEMENT_LIST | ε
WHILE_STMT ::= WHILE EXPRESSION DO STATEMENT_LIST END
EXPRESSION_STMT ::= EXPRESSION SEMICOLON
EXPRESSION ::= TERM ((ADD | SUB) TERM)*
TERM ::= FACTOR ((MUL | DIV) FACTOR)*
FACTOR ::= NUMBER | IDENTIFIER | (EXPRESSION)
ADD ::= +
SUB ::= -
MUL ::= *
DIV ::= /
NUMBER ::= [0-9]+
IDENTIFIER ::= [a-zA-ZçÇğĞıİöÖşŞüÜ_][a-zA-Z0-9çÇğĞıİöÖşŞüÜ_]*
IF ::= eğer
THEN ::= ise
ELSE ::= değilse
WHILE ::= dön
END ::= son!
ASSIGN ::= =
SEMICOLON ::= ;
LPAREN ::= (
RPAREN ::= )
COLON ::= :
EOF ::= end of file
