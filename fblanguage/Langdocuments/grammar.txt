PROGRAM ::= STATEMENT_LIST EOF

STATEMENT_LIST ::= STATEMENT (SEMICOLON STATEMENT)*

STATEMENT ::= ASSIGN_STMT | IF_STMT | WHILE_STMT | EXPRESSION_STMT

ASSIGN_STMT ::= VARIABLE = EXPRESSION SEMICOLON

VARIABLE ::= IDENTIFIER

IDENTIFIER ::= [a-zA-ZçÇğĞıİöÖşŞüÜ_][a-zA-Z0-9çÇğĞıİöÖşŞüÜ_]*

EXPRESSION_STMT ::= ASSIGN_STMT

EXPRESSION ::= TERM ((ADD | SUB) TERM)*
TERM ::= FACTOR ((MUL | DIV) FACTOR)*
FACTOR ::= VARIABLE | NUMBER | LPAREN EXPRESSION RPAREN
COMPARISON ::= EXPRESSION ((EQ | NEQ | LT | GT | LTE | GTE) EXPRESSION)
EQ ::= ==
NEQ ::= !=
LT ::= <
GT ::= >
LTE ::= <=
GTE ::= >=

NUMBER ::= [0-9]+

IF_STMT ::= IF_CLAUSE LBRACE THEN_STMT RBRACE 
IF_CLAUSE ::= IF LPAREN COMPARISON RPAREN
THEN_STMT ::= STATEMENT*


WHILE_STMT ::= WHILE_CLAUSE LBRACE DO_STMT RBRACE END_STMT
WHILE_CLAUSE ::= WHILE LPAREN COMPARISON RPAREN
DO_STMT ::= STATEMENT*
